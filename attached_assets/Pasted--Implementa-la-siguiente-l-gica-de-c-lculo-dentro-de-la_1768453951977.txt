export const calculateBOM = (inputs: ProjectInputs) => {
  let baseTier = "STR";
  let additionalItems = [];

  // 1. Lógica de Tier por Capacidad
  if (inputs.doors > 500 || inputs.users > 100000 || inputs.operators > 40) {
    baseTier = "ELT";
  } else if (inputs.doors > 128 || inputs.users > 50000 || inputs.operators > 20 || inputs.hasAdvancedAC) {
    baseTier = "ENT";
  } else if (inputs.doors > 32 || inputs.users > 1000 || inputs.operators > 10) {
    baseTier = "ADV";
  } else if (inputs.doors > 5 || inputs.users > 100 || inputs.operators > 1) {
    baseTier = "ESS";
  }

  // 2. Lógica de Add-ons (Ejemplo VMS)
  if (inputs.needsVideo) {
    additionalItems.push({
      id: "BIOSTARX-ADD-VMS",
      qty: inputs.videoChannels,
      name: "BioStar X Video Management License"
    });
  }

  // ... (Resto de la lógica de negocio)
  
  return { baseTier, additionalItems };
};
